#Reading the raw counts file
#Be sure you are set in the correct working directory
countData <- as.matrix(read.csv("gene_count_matrix.csv", row.names=1))
head(countData)

#Read the metadata file that dictates what conditions your sample names refer to
metadata <- read.csv("metadata.csv", row.names=1, stringsAsFactors = TRUE)
metadata

#Experimental design
designCondition <- ~ Replicate + Organoid + Condition
#designDonor <- ~ Replicate + Organoid + T_cell_Donor
ddsCondition <- DESeqDataSetFromMatrix(countData = countData,colData = metadata,design = designCondition)
ddsCondition <- DESeq(ddsCondition)
#You can also switch these to Donor later

#Make sure that counts were made for every sample
as.data.frame(colSums(countData))
#Check the size factors, which will account for changes in sequencing depth, library size, etc.
#A value of 0.55 means it will be scaled up by 0.55 in normalization; 1.19 means it will be scaled down by 1.19 in normalization
as.data.frame(sizeFactors(ddsCondition))
#Determine how many genes were lowly expressed; here the threshold is 10 counts and see how many will remain after this threshold
dim(ddsCondition)
dim(ddsCondition[rowSums(counts(ddsCondition)) > 10, ])
# Filter out lowly expressed genes
ddsCondition <- ddsCondition[rowSums(counts(ddsCondition)) > 10, ]

#Log transform the data to prepare it for analysis
rld <- rlogTransformation(ddsCondition, blind=FALSE)
head(assay(rld))
#vst <- vst(dds, blind=FALSE)
#head(assay(vst))

plotPCA(rld, intgroup="Condition", ntop = 5000)

#Update the PCA to show confidence intervals 
pcaData <- plotPCA(rld, intgroup = "Condition", ntop = 5000, returnData = TRUE)
#Calculate percent variance
percentVar <- round(100 * attr(pcaData, "percentVar"))
#Plot with ggplot2 and add confidence ellipses
ggplot(pcaData, aes(x = PC1, y = PC2, color = Condition)) +
  geom_point(size = 2) +
  stat_ellipse(level = 0.95) +  # 95% confidence ellipse
  xlab(paste0("PC1: ", percentVar[1], "% variance")) +
  ylab(paste0("PC2: ", percentVar[2], "% variance")) +
  theme_minimal() +
  coord_fixed()
#Check how each PC contributes to the variance
pca <- prcomp(t(assay(rld)))  # assumes rld is your rlog-transformed DESeqDataSet
#Extract and clean up variance proportions
prop_var <- summary(pca)$importance["Proportion of Variance", ] * 100
prop_var <- prop_var[is.finite(prop_var) & !is.na(prop_var)]
barplot(prop_var,
        main = "Proportion of Variance by Principal Component",
        ylab = "% Variance",
        xlab = "PC",
        col = "steelblue")

#Extract the loadings (gene contributions to each PC)
loadings <- pca$rotation
#Extract top 50 genes based on absolute loadings for the first PC (or another PC if needed)
top_genes <- names(sort(abs(loadings[,1]), decreasing = TRUE)[1:50])
#Subset the loadings matrix for these top 50 genes
top_loadings <- loadings[top_genes, ]
#Create the heatmap
heatmap <- pheatmap(top_loadings, 
                    cluster_cols = TRUE, 
                    show_rownames = TRUE, 
                    main = "Heatmap of Top 50 Genes' PCA Loadings",
                    color = colorRampPalette(c("blue", "white", "red"))(50))

#Run differential expression where p<0.05
res <- results(ddsCondition, alpha=0.05)
head(res)
#Perform LFC shrinkage
res <- lfcShrink(ddsCondition, res=res, type="ashr")
plotMA(res, main="Shrunken FC")
#Remove any duplicated items
res2<-res[!duplicated(res),]
#Split the columns 
library(stringr)
res2$ENSEMBL=as.data.frame(str_split(rownames(res2), "\\..*", simplify = TRUE))$V1
res2$GENEID=as.data.frame(str_split(rownames(res2), ".*\\|", simplify = TRUE))$V2
head(res2)
#Load the gene annotation 
library(org.Hs.eg.db)
#Extract gene information (e.g., gene symbols and biotypes)
gene_info <- AnnotationDbi::select(org.Hs.eg.db, keys=keys(org.Hs.eg.db, keytype="ENSEMBL"), columns=c("SYMBOL", "GENETYPE"), keytype="ENSEMBL")
#Convert to data frame
gene_info <- as.data.frame(gene_info)
head(gene_info)
#Merge DESeq2 results with gene annotations
res2<-as.data.frame(res2)
merged_results <- merge(res2, gene_info, by.x="ENSEMBL", by.y="ENSEMBL")
head(merged_results)
#Subset non-coding out
merged_results=merged_results[merged_results$GENETYPE=="protein-coding",]

#Function to map ENSEMBL IDs to gene symbols for a core_enrichment string
map_core_genes <- function(core_enrich_str) {
  genes <- strsplit(core_enrich_str, split = "/")[[1]]
  mapped <- mapIds(org.Hs.eg.db,
                   keys = genes,
                   column = "SYMBOL",
                   keytype = "ENSEMBL",
                   multiVals = "first")
  paste(na.omit(mapped), collapse = ", ")
}
